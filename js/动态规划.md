#### 递归+动态规划：

li = [2,3,5,7,9]，输出任意组合，可以重复选，输出所有和是 13 的组合。

```js
var li = [2, 3, 5, 7, 9]

function getCombination(target, arr = []) {
li.forEach(value => {
if (target - value < 0) {
return
}
if (target - value === 0) {
console.log(arr.concat(value))
return
}
getCombination(target - value, arr.concat(value))
})
}

getCombination(13)
```
